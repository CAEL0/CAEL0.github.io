import sys


def dfs(cur):
    visit[cur] = 1
    for nxt in graph[cur]:
        if (b[nxt] == -1) or (not visit[b[nxt]] and dfs(b[nxt])):
            b[nxt] = cur
            return 1
    return 0


n, m, k = map(int, sys.stdin.readline().split())
graph = [0] * (2 * n)

for i in range(n):
    data = list(map(int, sys.stdin.readline().split()))
    graph[2 * i] = data[1:]
    graph[2 * i + 1] = data[1:]

b = [-1] * (m + 1)
ans1 = 0
for i in range(n):
    visit = [0] * (2 * n)
    ans1 += dfs(2 * i)

ans2 = 0
for i in range(n):
    visit = [0] * (2 * n)
    ans2 += dfs(2 * i + 1)
    if ans2 == k:
        break

print(ans1 + ans2)
